    • func main() {
	//容量为3
	var strCha = make(chan string, 3)
	strCha <- "aaa"
	strCha <- "bbb"
	//strCha <- "ccc"
	fmt.Println(len(strCha)) //2
	fmt.Println(cap(strCha)) //3
	str := <-strCha
	fmt.Println(str)         // aaa
	fmt.Println(len(strCha)) //1
	fmt.Println(cap(strCha)) //3
	//关闭channel，不能往里面写数据了，但是还是可以往里面读数据
	close(strCha)

}
